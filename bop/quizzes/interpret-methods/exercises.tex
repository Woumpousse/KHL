class Circle {
    private double radius;

    public void getRadius() {
        return radius;
    }

    public void setRadius(double newRadius) {
        radius = newRadius;
    }

    public double area() {
        return radius * radius * 3.14;
    }
}

class App {
    public static void main(String[] args) {
        Circle c = new Circle();
        
        System.out.println(c.getRadius());
    }
}
