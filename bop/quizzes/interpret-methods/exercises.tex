class Averager {
    private int sum;
    private int count;

    public void add(int x) {
        sum += x;
        count++;
    }

    public int getAverage() {
        return sum / count;
    }
}

class App {
    public static void main(String[] args) {
        Averager avg = new Averager();

        avg.add(1);
        avg.add(2);
        avg.add(3);
        avg.add(4);
        avg.add(5);

        System.out.println( avg.getAverage() );
    }
}
\end{lstlisting}\clearpage

class Foo {
    private int x;
    private int y;

    public void set(int a, int b) {
        x = a;
        y = b;
    }

    public void bar() {
        int t = x;
        x = y;
        y = t;
    }

    public int getX() {
        return x;
    }

    public int getY() {
        return y;
    }
}

class App {
    public static void main(String[] args) {
        Foo foo = new Foo();

        foo.set(3, 5);
        foo.bar();

        System.out.println( foo.getX() + "," + foo.getY() );
    }
}
