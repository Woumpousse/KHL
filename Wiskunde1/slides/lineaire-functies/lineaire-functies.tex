        }

        \only<2>{
          \draw[plot] plot (\x, 2 * \x);

\begin{frame}
  \frametitle{Richtingsco\"efficient $m$: Wiskundig Bewijs}
  \[
     \begin{array}{rcl}
       f(x + 1) - f(x) & = & (m \cdot (x + 1) + q) - (m \cdot x + q) \\
                       & = & m \cdot (x + 1) + q - m \cdot x - q \\
                       & = & m \cdot (x + 1) - m \cdot x \\
                       & = & m \cdot x + m - m \cdot x \\
                       & = & m \\
     \end{array}
  \]
\end{frame}
