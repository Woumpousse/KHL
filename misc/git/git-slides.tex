\newcommand{\bits}[1]{
  {
    \pgfmathparse{bin(128+mod(#1 * 229 + 73, 128))}\let\x\pgfmathresult
    \pgfmathparse{bin(128+mod(#1 * 541 + 111, 128)}\let\y\pgfmathresult
    \parbox{.9cm}{\centering\tt \x \\ \y}
  }
}
